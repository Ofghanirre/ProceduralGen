107void _interpolateColor(int value, int max, float midThreshold, int minR, int minG, int minB, int maxR, int maxG, int maxB, int& R, int& G, int& B) {
108 // Calculer le facteur d'interpolation
109 double factor = static_cast<double>(value) / max;
110 if (factor < midThreshold) {
111 factor = factor * (1. - midThreshold);
112 } else {
113 factor = (factor - midThreshold) / (1. - midThreshold);
114 }
115 // Interpoler chaque composante de couleur
116 R = minR + static_cast<int>(factor * (maxR - minR));
117 G = minG + static_cast<int>(factor * (maxG - minG));
118 B = minB + static_cast<int>(factor * (maxB - minB));
119}
