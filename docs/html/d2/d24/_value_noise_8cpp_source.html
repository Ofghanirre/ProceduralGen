65 float eval(Vec2f p) const
66 {
67 int xi = std::floor(p.x);
68 int yi = std::floor(p.y);
69
70 float tx = p.x - xi;
71 float ty = p.y - yi;
72
73 int rx0 = xi & kMaxTableSizeMask;
74 int rx1 = (rx0 + 1) & kMaxTableSizeMask;
75 int ry0 = yi & kMaxTableSizeMask;
76 int ry1 = (ry0 + 1) & kMaxTableSizeMask;
77
78 // random values at the corners of the cell using permutation table
79 const float & c00 = r[permutationTable[permutationTable[rx0] + ry0]];
80 const float & c10 = r[permutationTable[permutationTable[rx1] + ry0]];
81 const float & c01 = r[permutationTable[permutationTable[rx0] + ry1]];
82 const float & c11 = r[permutationTable[permutationTable[rx1] + ry1]];
83
84 // remapping of tx and ty using the Smoothstep function
85 float sx = smoothstep(tx);
86 float sy = smoothstep(ty);
87
88 // linearly interpolate values along the x axis
89 float nx0 = lerp(c00, c10, sx);
90 float nx1 = lerp(c01, c11, sx);
91
92 // linearly interpolate the nx0/nx1 along they y axis
93 return lerp(nx0, nx1, sy);
94 }
